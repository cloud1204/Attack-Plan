  \item $1 \leq n, m \leq 10^5$。
  \item $1 \leq b_i \leq a_i \leq 10^{9}$（$1 \leq i \leq n$）。
  \item $1 \leq w_i \leq 10^4$（$1 \le i \leq m$）。
  \item $1 \leq u_i, v_i \leq n$（$1 \leq i \leq m$）。
  \item $u_i \neq v_i$（$1 \leq i \leq m$）。
  \item 所有輸入都是整數。
\end{itemize}
